#include <stdio.h>
int main() {

    int n,arr[100][100],total1=0,total2=0;
    scanf("%d",&n);
    for(int i=0;i<n;++i)
        {
        for(int j=0;j<n;++j){

            scanf("%d",&arr[i][j]);
             if(i==j) total1+=arr[i][j];
              if((i+j)==(n-1)) total2+=arr[i][j];

        }
    }
     if(total1>total2) printf("%d",(total1-total2));
     else printf("%d",(total2-total1));


    return 0;
}
